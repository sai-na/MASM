PRINT MACRO MSG
 LEA DX, MSG
 MOV AH, 09H
 INT 21H
ENDM
.MODEL SMALL
.STACK
.DATA
 MSG1 DB 0AH,0DH,"ENTER THE STRING: $"
 MSG2 DB 0AH,0DH,"SORTED STRING IS: $"
 STR1 DB 50 DUP ("$")
 LEN1 DB 0
.CODE
START:
 MOV AX, @DATA
 MOV DS, AX
 PRINT MSG1
 LEA DX, STR1
 MOV AH, 0AH
 INT 21H
 MOV AL, STR1+1
 MOV LEN1, AL
 MOV DL, AL
 DEC DL
 ; Bubble sort
 J1:
 LEA SI, STR1+2 ;
 XOR CX, CX
 MOV CL, DL
 J3:
 MOV AL, [SI]
 CMP AL, [SI+1]
 JBE J2
 MOV AH, [SI+1]
 MOV [SI+1], AL
 MOV [SI], AH
 J2:
 INC SI
 LOOP J3
 DEC DL
 CMP DL, 00
 JNZ J1
 PRINT MSG2
 PRINT STR1+2 ;
 MOV AH, 4CH
 INT 21H
END START
