.MODEL SMALL
.STACK
.DATA
 MSG1 DB 0AH,0DH,"ENTER THE STRING: $"
 MSG2 DB 0AH,0DH,"REVERSED STRING IS: $"
 STR1 DB 100 DUP('$')
.CODE
START:
 MOV AX,@DATA
 MOV DS,AX
 LEA DX,MSG1
 MOV AH,09H
 INT 21H
 LEA DX,STR1
 MOV AH,3FH
 INT 21H
 LEA DX,MSG2
 MOV AH,09H
 INT 21H
 MOV SI,00H
L2:
 CMP STR1[SI],"$"
 JE L1
 INC SI
 JMP L2
L1:
 DEC SI
 MOV DL,STR1[SI]
 MOV AH,02H
 INT 21H
 JNZ L1
 MOV AH,4CH
 INT 21H
END START
