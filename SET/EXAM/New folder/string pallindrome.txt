.MODEL SMALL
.STACK
.DATA
 D1 DB 100 DUP("$")
 D2 DB 0AH,0DH,"ENTER THE STRING: $"
 D3 DB 0AH,0DH,"STRING IS PALINDROME $"
 D4 DB 0AH,0DH,"NOT PALINDROME $"
.CODE
START:
 MOV AX,@DATA
 MOV DS,AX
 LEA DX,D2
 MOV AH,09H
 INT 21H
 LEA DX,D1
 MOV AH,3FH
 INT 21H
 SUB AX,02H
 MOV SI,AX
 DEC SI
 MOV CX,AX
 MOV DI,00H
L1:
 MOV AH,D1[DI]
 CMP AH,D1[SI]
 JNE L2
 INC DI
 DEC SI
 LOOP L1
 LEA DX,D3
 MOV AH,09H
 INT 21H
 JMP L3
L2:
 LEA DX,D4
 MOV AH,09H
 INT 21H
L3:
 MOV AH,4CH
 INT 21H
END START
